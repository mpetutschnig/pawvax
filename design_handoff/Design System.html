06 — Implementation

index.css Root Variables

Drop this into pwa/src/index.css replacing the existing :root {} block.

pwa/src/index.css — :root variables 
/* ═══ VAX.PET DESIGN SYSTEM TOKENS ═══ */
:root {
  /* Primary (Trust Blue — swap hue for other variants) */
  --primary-50:  oklch(96% 0.04 240);
  --primary-100: oklch(91% 0.08 240);
  --primary-200: oklch(82% 0.12 240);
  --primary-300: oklch(68% 0.16 240);
  --primary-400: oklch(55% 0.20 240);
  --primary-500: oklch(44% 0.22 240);  /* default interactive */
  --primary-600: oklch(38% 0.20 240);  /* hover state */
  --primary-700: oklch(32% 0.18 240);
  --primary-800: oklch(24% 0.14 240);
  --primary-900: oklch(16% 0.10 240);

  /* Accent (Teal) */
  --accent-400:  oklch(68% 0.15 195);
  --accent-500:  oklch(56% 0.17 195);
  --accent-600:  oklch(46% 0.15 195);

  /* Semantic */
  --success-50:  oklch(95% 0.06 145);
  --success-500: oklch(52% 0.18 145);
  --success-600: oklch(43% 0.16 145);
  --warning-50:  oklch(96% 0.06 80);
  --warning-500: oklch(68% 0.20 80);
  --warning-600: oklch(57% 0.18 80);
  --danger-50:   oklch(96% 0.05 25);
  --danger-500:  oklch(56% 0.22 25);
  --danger-600:  oklch(46% 0.20 25);

  /* Surfaces (Light) */
  --bg:            oklch(97.5% 0.007 240);
  --bg-elevated:  oklch(100%  0     0);
  --surface:       oklch(95%   0.010 240);
  --border:        oklch(87%   0.014 240);
  --text-primary: oklch(16%   0.025 240);
  --text-secondary:oklch(46%  0.025 240);
  --text-tertiary:oklch(66%   0.018 240);

  /* Shadows */
  --shadow-sm:  0 1px 3px oklch(0% 0 0 / 0.07);
  --shadow-md:  0 4px 12px oklch(0% 0 0 / 0.09);
  --shadow-lg:  0 8px 24px oklch(0% 0 0 / 0.11);
  --shadow-xl:  0 16px 48px oklch(0% 0 0 / 0.13);
  --glass-bg:   oklch(100% 0 0 / 0.72);

  /* Spacing */
  --space-1: 4px;  --space-2: 8px;  --space-3: 12px;
  --space-4: 16px; --space-5: 20px; --space-6: 24px;
  --space-8: 32px; --space-10:40px; --space-12:48px;

  /* Radii */
  --radius-xs:   4px;  --radius-sm:  8px;
  --radius-md:   12px; --radius-lg:  16px;
  --radius-xl:   20px; --radius-2xl: 24px;
  --radius-full: 9999px;

  /* Typography */
  --font-display: 'Outfit', system-ui, sans-serif;
  --font-body:    'DM Sans', system-ui, sans-serif;
  --font-mono:    'DM Mono', 'Courier New', monospace;

  /* Transitions */
  --ease-out: cubic-bezier(0.16, 1, 0.3, 1);
  --t-fast:  120ms; --t-base: 200ms; --t-slow: 350ms;
}
